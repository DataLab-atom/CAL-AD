def compute_omega_v2(t: int, n: int, r0: float, rho: float, M: float, L: float) -> float:
    """Compute the scaling parameter omega with a more dynamic decay strategy."""
    if t % n == 0:
        decay_factor = rho**(t // n)
        dynamic_scaling = np.exp(-M * np.sqrt(L) * decay_factor)
        return (1 + r0 * dynamic_scaling)**2
    return 1.0
